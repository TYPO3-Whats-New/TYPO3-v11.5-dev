% Feature | 95176 | Introduce <f:transformhtml> view helper

\begin{frame}[fragile]
	\frametitle{Extbase et Fluid}
	\framesubtitle{Fluid ViewHelpers (2)}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item Le ViewHelper suivant est introduit~:\newline
			\small\texttt{<f:transform.html>}\normalsize
		\item Ce ViewHelper résoud et transforme les gestionnaires de lien TYPO3 comme
			les adresses \texttt{t3://}
		\item Il est indépendant de toute configuration TypoScript et travail
			sans mélanger les sujet avec \small\texttt{ContentObjectRenderer::parseFunc}
\begin{lstlisting}
<f:transform.html selector="a.href,div.data-uri">
  <a href="t3://page?uid=1" class="page">visit</a>
  <div data-uri="t3://page?uid=1" class="page trigger">visit</div>
</f:transform.html>
\end{lstlisting}
	\end{itemize}

\end{frame}
