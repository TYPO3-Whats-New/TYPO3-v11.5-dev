% Important | 95298 | Fluid Viewhelpers will be declared final in v12

\begin{frame}[fragile]
	\frametitle{Extbase et Fluid}
	\framesubtitle{Fluid ViewHelpers (1)}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Tous les classes ViewHelper Fluid seront déclarées «~\texttt{final}~» dans
			TYPO3 v12
		\item Les extensions ne peuvent plus les étendre avec leurs propres variantes
		\item Les développeurs d'extension sont invités à utiliser les classes abstraites
			fournies par Fluid (si elles ne sont pas marquées \texttt{\@internal})
		\item Les développeurs sont encouragés à déjà effectuer les ajustements appropriés à
			leurs ViewHelpers sous TYPO3 v11
	\end{itemize}

\end{frame}
