% TCA option "groupName" across Core TCA config
% https://review.typo3.org/c/Packages/TYPO3.CMS/+/71316

\begin{frame}[fragile]
	\frametitle{Changements pour les intégrateurs et développeurs}
	\framesubtitle{Option TCA \texttt{groupName}}

	\begin{itemize}
		\item Le noyau de TYPO3 fait usage de l'option TCA «~\texttt{groupName}~»
		\item Elle permet de grouper les enregistrements dans l'«~assistant nouvel enregistrement~»
		\item Ces mots clés sont utilisés par le noyau~:\newline
			\small
				\texttt{system}, \texttt{frontendaccess}, \texttt{backendaccess}, and \texttt{content}
			\normalsize
		\item Les extensions qui fournissent leurs propres groupes et mots clés continuent de fonctionner
	\end{itemize}

\end{frame}
