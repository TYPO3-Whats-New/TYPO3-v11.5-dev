% Feature | 94868 | Introduce Bootstrap 5 compatible and accessible templates

\begin{frame}[fragile]
	\frametitle{Changements pour les intégrateurs et développeurs}
	\framesubtitle{Gabarits compatibles avec Bootstrap 5}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Des gabarits compatible avec Bootstrap version 5 sont ajoutés à
			Form Framework
		\item Ces variantes sont enregistrées dans\newline
			\small\texttt{EXT:form/Resources/Private/Frontend/Version2/}\normalsize
		\item Dans la plupart des cas, seul la définition des classes CSS appropriées est nécessaire
			pour personnaliser l'apparence des gabarits
		\item Cependant, TYPO3 continue d'utiliser les anciennes variantes par défaut
			pour éviter de casser les configurations existantes
		\item Les nouveaux gabarits doivent être activés explicitement
		\item Voir le \href{https://docs.typo3.org/c/typo3/cms-core/master/en-us/Changelog/11.5/Feature-94868-IntroduceBootstrap5CompatibleAndAccessibleTemplates.html}{journal des changements (en)}
			pour plus de détails
	\end{itemize}

\end{frame}
