% Important | 95384 | TCA internal_typedb optional for typegroup

\begin{frame}[fragile]
	\frametitle{Changements pour les intégrateurs et développeurs}
	\framesubtitle{Option TCA \texttt{internal\_type}}

	\begin{itemize}
		\item Lorsque le type TCA «~\texttt{group}~» est utilisé, l'option
			«~\texttt{internal\_type}~» défini quel type d'enregistrement peut
			être référencé\newline
			\small(les valeurs valides sont «~\texttt{folder}~» et «~\texttt{db}~»)\normalsize
		\item Comme «~\texttt{db}~» est souvent utilisé, TYPO3 utilise cette valeur par défaut
		\item Les développeurs d'extension peuvent omettre l'option
			«~\texttt{internal\_type}~» dans la majorité des cas
	\end{itemize}

\end{frame}
