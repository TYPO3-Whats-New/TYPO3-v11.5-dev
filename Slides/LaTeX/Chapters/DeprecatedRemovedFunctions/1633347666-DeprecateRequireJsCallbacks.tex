% Deprecation | 95200 | Deprecate RequireJS callbacks as inline JavaScript

\begin{frame}[fragile]
	\frametitle{Fonctions dépréciées/retirées}
	\framesubtitle{Fonctions de rappel RequireJS}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Utiliser les fonctions de rappel avec RequireJS est marqué
			\textbf{déprécié}\newline
			\small(par exemple~: \texttt{\$resultArray['requireJsModules']} avec une \texttt{string} litérale)\normalsize
		\item Ces appels doivent être remplacés par les nouvelles déclarations \texttt{JavaScriptModuleInstruction}
		\item Voir le \href{https://docs.typo3.org/c/typo3/cms-core/master/en-us/Changelog/11.5/Deprecation-95200-DeprecateRequireJSCallbacksAsInlineJavaScript.html}{journal des changements (en)}
			pour plus de détail et les options de migration
	\end{itemize}

\end{frame}
