% Deprecation | 95351 | Custom JSWindow options in HMENU settings

\begin{frame}[fragile]
	\frametitle{Fonctions dépréciées/retirées}
	\framesubtitle{Options JSWindow}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{7}{9}\ttfamily}

	\begin{itemize}
		\item Les options de \texttt{HMENU}/\texttt{TMENU} pour ouvrir les liens
			du menu dans une pop-up JavaScript personnalisée sont marquées \textbf{dépréciées}~:
\begin{lstlisting}
# deprecated:
page.123 = HMENU
page.123.1 = TMENU
page.123.1.JSWindow = 1
page.123.1.JSWindow.params = width=200,height=300,status=0,menubar=0
\end{lstlisting}
		\item Accéder à une page frontend avec \texttt{HMENU} et les popus \texttt{JSwindow}
			déclenchera un avertissement de dépréciation
	\end{itemize}

\end{frame}
