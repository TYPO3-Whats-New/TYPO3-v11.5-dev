% Deprecation | 95318 | TypoScript parseFunc.sword

\begin{frame}[fragile]
	\frametitle{Fonctions dépréciées/retirées}
	\framesubtitle{TypoScript \texttt{parseFunc.sword}}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item L'option TypoScript \texttt{parseFunc.sword} permet d'encapsuler les
			mots de recherche spécifiquement lorsque \texttt{no\_cache = 1} est définit
		\item Les fonctionnalité est dépréciée
		\item Les installations sont affectées si elle permettent un lien comme~:\newline
			\smaller\texttt{https://example.com/?no\_cache=1\&sword\_list\%5B\%5D=search}\normalsize
	\end{itemize}

\end{frame}
