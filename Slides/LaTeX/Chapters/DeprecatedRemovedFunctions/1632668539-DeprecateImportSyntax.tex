% Deprecation | 95349 | TypoScript: page.includeCSS.import
% Deprecation | 95349 | TypoScript: page.includeCSSLibs.import

\begin{frame}[fragile]
	\frametitle{Fonctions dépréciées/retirées}
	\framesubtitle{Inclusion de CSS externe}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{7}{9}\ttfamily}

	\begin{itemize}
		\item L'option d'utilisation de la syntaxe \texttt{@import} pour inclure des fichiers CSS
			externes par TypoScript est marquée \textbf{dépréciée}~:
\begin{lstlisting}
# deprecated:
page = PAGE
page.includeCSSLibs.file1 = fileadmin/benni.css
page.includeCSSLibs.file1.import = 1
\end{lstlisting}
		\item Options de migration~:
			\begin{itemize}
				\item utiliser la balise \texttt{<link>} ou
				\item crée manuellement une inclusion CSS en TypoScript
			\end{itemize}
	\end{itemize}

\end{frame}
